${text}
`; chatHistory.push({role:'user',content:text}); const tid='t'+Date.now(); msgs.innerHTML+=`
🌿
Ember
`; msgs.scrollTop=msgs.scrollHeight; // Abort the request after 20s so the typing dots never spin forever const controller=new AbortController(); const timeoutId=setTimeout(()=>controller.abort(),20000); try{ const res=await fetch('/api/chat',{ method:'POST', headers:{'Content-Type':'application/json'}, body:JSON.stringify({system:SYS+'\n\nUSER DATA:\n'+aiContext,messages:chatHistory.slice(-10)}), signal:controller.signal }); clearTimeout(timeoutId); const data=await res.json(); // If the server returned an error, SHOW it instead of pretending it worked if(!res.ok){ console.error('Advisor API error:',res.status,data); const tEl=document.getElementById(tid);if(tEl)tEl.remove(); const errText=data.error||'Something went wrong. Please try again.'; msgs.innerHTML+=`
🌿
Ember
${errText}

`; msgs.scrollTop=msgs.scrollHeight; return; } const reply=data.content?.map(b=>b.text||'').join('')||'Let me think about that…'; const tEl=document.getElementById(tid);if(tEl)tEl.remove(); chatHistory.push({role:'assistant',content:reply}); msgs.innerHTML+=`
🌿

${reply.replace(/\n/g,'

`; }catch(err){ clearTimeout(timeoutId); console.error('Advisor fetch failed:',err); const tEl=document.getElementById(tid);if(tEl)tEl.remove(); const m=err.name==='AbortError' ?'That took too long to respond. 🌿

`; } msgs.scrollTop=msgs.scrollHeight; }
